OpenClaw has been deployed to namespace {{ .Release.Namespace }}.

1. Get the application URL:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "openclaw.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export OPENCLAW_URL="http://$NODE_IP:$NODE_PORT/"
  echo $OPENCLAW_URL
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "openclaw.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "openclaw.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  export OPENCLAW_URL="http://$SERVICE_IP:{{ .Values.service.port }}/"
  echo $OPENCLAW_URL
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "openclaw.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export OPENCLAW_URL="http://127.0.0.1:{{ .Values.service.port }}/"
  echo "Visit $OPENCLAW_URL after port-forward:"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.service.port }}:{{ .Values.service.port }}
{{- end }}

{{- if eq (.Values.gateway.auth.mode | default "token") "token" }}

2. Fetch the Gateway token:
  export OPENCLAW_TOKEN=$(kubectl get secret -n {{ .Release.Namespace }} {{ include "openclaw.gatewayAuthSecretName" . }} -o jsonpath="{.data.{{ include "openclaw.gatewayAuthSecretKey" . }}}" | openssl base64 -d -A)
  echo "$OPENCLAW_URL?token=$OPENCLAW_TOKEN"
{{- end }}

{{- if eq .Values.gateway.auth.mode "password" }}

2. Fetch the Gateway password:
  kubectl get secret -n {{ .Release.Namespace }} {{ include "openclaw.gatewayAuthSecretName" . }} -o jsonpath="{.data.{{ include "openclaw.gatewayAuthSecretKey" . }}}" | openssl base64 -d -A
{{- end }}

3. Configure providers or web search inside the pod when needed:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "openclaw.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl exec -it --namespace {{ .Release.Namespace }} $POD_NAME -- openclaw configure --section web

{{- if .Values.agents }}

NOTE: Agent SOUL.md files are written to per-agent workspaces and are overwritten from chart values on every pod restart.
      To persist changes, update values.agents in your values file, not the pod filesystem.
{{- end }}
